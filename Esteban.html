import os
from tensorflow.keras.preprocessing.image import img_to_array #Import the function directly
import numpy as np
import cv2


#preparar función para procesar las imágenes

def process_image(image_path):
  img = cv2.imread(image_path) #read image path
  img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY) #convert it into grayscale
  img_array = img_to_array(img) #convert it into an array
  resized_img = cv2.resize(img_array, (100, 100)) #resize into a 100 x 100px
  normalized_img = resized_img / 255.0  #normalize the arrays
  return normalized_img


#PROCESAR IMÁGENES DE COVID
covid = [] #create an empty variable to store the arrays


for filename in os.listdir('/content/COVID'): #process every image
  if filename.endswith(".png"):
    image_path = os.path.join('/content/COVID', filename) #save image path
    processed_image = process_image(image_path) #process the image
    covid.append(processed_image) #store it into the "covid" variable


covid = np.array(covid) #convert it into an array



#PROCESAR IMÁGENES DE NO-COVID

    processed_image = process_image(image_path) #process the image
    no_covid.append(processed_image) #store it into the "covid" variable

Step 3. Data sampling¶

  1. Flatten the arrays: (nrow, 100, 100) to (nrow, 10000)
  2. Create an output column (0 = no covid, 1 = covid)
  3. Split the datasets into X (pixels), and Y (output)
  4. Stratified sampling (to guarantee representative data in both training and test sets) into a training set (80%) and a test set (20%).
In [5]:
# Add a column of 1s to the covid array
covid_with_label = np.c_[covid.reshape(covid.shape[0], -1), np.ones(covid.shape[0])[:, np.newaxis]]

# Add a column of 0s to the no_covid array
nocovid_with_label = np.c_[no_covid.reshape(no_covid.shape[0], -1), np.zeros(no_covid.shape[0])[:, np.newaxis]]

# Concatenate the two arrays vertically
df = np.concatenate((covid_with_label, nocovid_with_label), axis=0)


#stratified sampling
from sklearn.model_selection import StratifiedShuffleSplit

# Assuming the last column of 'df' contains the labels (1 for COVID, 0 for non-COVID)
X = df[:, :-1]  # Features (image data)
y = df[:, -1]   # Labels

# Perform stratified sampling
sss = StratifiedShuffleSplit(n_splits=1, test_size=0.2, random_state=42)  # Adjust test_size as needed
for train_index, test_index in sss.split(X, y):
  X_train, X_test = X[train_index], X[test_index]
  y_train, y_test = y[train_index], y[test_index]

# Now you have stratified samples in X_train, X_test, y_train, and y_test

print("Se hizo un muestreo estratificado")
print(f'Training set (80%): {X_train.shape}')
print(f'Test set (20%): {X_test.shape}')

from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Flatten, Dropout



NN = Sequential([
    Dense(128, activation='relu'),
    Dense(64, activation='relu'),
    Flatten(), # Flatten the output before the final Dense layer
    Dropout(0.5),
    Dense(1, activation='sigmoid')
])

NN.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

NN.fit(X_train, y_train, epochs=10, batch_size=32, validation_data=(X_test, y_test))



from sklearn.metrics import confusion_matrix

from sklearn.model_selection import KFold
from sklearn.metrics import accuracy_score

# Initialize lists to store accuracy scores
nn_scores = []
svm_scores = []
rf_scores = []
lr_scores = []

# Set up K-fold cross-validation
kf = KFold(n_splits=10, shuffle=True, random_state=42)

# Iterate through folds
for train_index, val_index in kf.split(X):
    X_train, X_val = X[train_index], X[val_index]
    y_train, y_val = y[train_index], y[val_index]

    # Neural Network
    y_pred_nn = (NN.predict(X_val) > 0.5).astype(int)
    nn_scores.append(accuracy_score(y_val, y_pred_nn))

    # Support Vector Machine

    y_pred_svm = SVM.predict(X_val)
    svm_scores.append(accuracy_score(y_val, y_pred_svm))

    # Random Forest

    y_pred_rf = RF.predict(X_val)
    rf_scores.append(accuracy_score(y_val, y_pred_rf))

    # Logistic Regression

    y_pred_lr = LR.predict(X_val)
    lr_scores.append(accuracy_score(y_val, y_pred_lr))

# Plot accuracy scores for each fold
plt.figure(figsize=(10, 6))
plt.plot(range(1, 11), nn_scores, marker='o', label='Neural Network')
plt.plot(range(1, 11), svm_scores, marker='o', label='Support Vector Machine')
plt.plot(range(1, 11), rf_scores, marker='o', label='Random Forest')
plt.plot(range(1, 11), lr_scores, marker='o', label='Logistic Regression')
plt.xlabel('Fold')
plt.ylabel('Accuracy')
plt.title('K-Fold Cross Validation Accuracy')
plt.legend()
plt.grid(True)
plt.show()
